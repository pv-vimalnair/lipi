* Apple Liquid Glass — CSS tokens (port from WWDC25)

:root {
/* === Glass surfaces === */
--glass-bg: rgba(255, 255, 255, 0.55);
--glass-bg-strong: rgba(255, 255, 255, 0.72);
--glass-bg-subtle: rgba(255, 255, 255, 0.40);
--glass-bg-tinted: rgba(37, 99, 235, 0.14);
/* === Glass edge (squircle + inner highlights) === */
--glass-border: 0.5px solid rgba(255, 255, 255, 0.65);
--glass-highlight: inset 0 1px 0 rgba(255, 255, 255, 0.7);
--glass-depth: inset 0 -1px 0 rgba(0, 0, 0, 0.04);
--glass-outer-shadow: 0 12px 32px rgba(15, 15, 18, 0.08);
--glass-specular: linear-gradient(180deg, ...);
/* === Glass blur (illumination) === */
--glass-blur: blur(30px) saturate(180%);
--glass-blur-subtle: blur(15px) saturate(180%);
/* === Geometry (squircle) === */
--radius-glass: 24px;
--radius-control: 12px;
--radius-pill: 999px;
/* === Updated for light theme === */
--color-bg-base: linear-gradient(180deg, #FFF, #F8F9FC);
--color-bg-pane: #FFFFFF; /* Monaco needs solid */
--color-text-primary: #14171E;
--color-text-secondary: #5C5C66;
--color-accent: #2563EB;
}
/* Auto light/dark via media query — no JS hook needed */

--glass-bg: rgba(20, 20, 28, 0.55);
}
